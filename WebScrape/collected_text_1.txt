import pyautogui
import pyperclip
import time
from wordcloud import WordCloud
import matplotlib.pyplot as plt
import os

# Function to copy text from the active Chrome tab
def copy_text_from_tab():
    # Select all text (Command+A for Mac)
    pyautogui.hotkey('command', 'a')
    time.sleep(2)  # Small delay to ensure action is registered
    # Copy the selected text (Command+C for Mac)
    pyautogui.hotkey('command', 'c')
    time.sleep(2)  # Wait for clipboard to update
    # Get text from clipboard
    return pyperclip.paste()

# Function to generate a unique filename if the file already exists
def get_unique_filename(base_name):
    counter = 1
    filename = f"{base_name}.txt"
    while os.path.exists(filename):
        filename = f"{base_name}_{counter}.txt"
        counter += 1
    return filename

# Define words to ignore in the word cloud
ignore_words = {
    'the', 'and', 'to', 'of', 'a', 'in', 'that', 'it', 'is', 'was', 'i', 'for', 'on', 
    'you', 'he', 'with', 'as', 'at', 'his', 'by', 'an', 'be', 'this', 'which', 'or', 
    'from', 'but', 'not', 'are', 'they', 'we', 'all', 'can', 'one', 'your', 'their', 
    'has', 'more', 'have', 'will', 'if', 'about', 'no', 'my', 'so', 'what'
}

# Initialize an empty string to accumulate text from all tabs
all_text = ""

# Number of tabs you want to process
num_tabs = 11  # Set to the actual number of tabs you have open

# Ensure Chrome is active window before running the loop
print("Switch to Chrome and ensure it is the active window. You have 5 seconds...")
time.sleep(5)

# Iterate over each tab to grab the content
for i in range(num_tabs):
    # Copy text from the current tab
    tab_text = copy_text_from_tab()
    all_text += tab_text + "\n"
    # Switch to the next tab (Command + Option + Right Arrow)
    pyautogui.hotkey('command', 'option', 'right')  # Adjust if needed
    time.sleep(2)  # Wait a moment to switch tabs

# Save all text to a unique file
filename = get_unique_filename('collected_text')
with open(filename, 'w', encoding='utf-8') as f:
    f.write(all_text)

# Generate and display the word cloud, ignoring specified words
wordcloud = WordCloud(width=1500, height=900, background_color='white', stopwords=ignore_words).generate(all_text)

plt.figure(figsize=(12, 6))
plt.imshow(wordcloud, interpolation='bilinear')
plt.axis('off')
plt.show()

print(f"Text saved to {filename}")

‘SNL’ Sets Maya Rudolph and Jake Gyllenhaal as Hosts With Vampire Weekend and Sabrina Carpenter as Musical Guests

By Selome Hailu

Plus Icon
SNL
Getty
“Saturday Night Live” has set Maya Rudolph to host its May 11 episode alongside Vampire Weekend as musical guest, while the May 18 episode will by headlined by Jake Gyllenhaal with performances from Sabrina Carpenter.

Rudolph was an “SNL” cast member for seven seasons from 2000 to 2007 and returns to the late-night sketch show for her third time as host, having previously hosted in 2012 with musical guest Sleigh Bells and in 2021 alongside Jack Harlow. Currently, she stars in and executive produces the Apple TV+ comedy “Loot.” Besides “SNL,” her prominent credits include “Bridesmaids,” “Sisters,” “The Good Place” and “Big Mouth.”

Plus Icon
SNL

“Saturday Night Live” has set Maya Rudolph to host its May 11 episode alongside Vampire Weekend as musical guest, while the May 18 episode will by headlined by Jake Gyllenhaal with performances from Sabrina Carpenter.

Rudolph was an “SNL” cast member for seven seasons from 2000 to 2007 and returns to the late-night sketch show for her third time as host, having previously hosted in 2012 with musical guest Sleigh Bells and in 2021 alongside Jack Harlow. ‘SNL’ Sets Maya Rudolph and Jake Gyllenhaal as Hosts With Vampire Weekend and Sabrina Carpenter as Musical Guests

By Selome Hailu

Plus Icon
SNL
Getty
“Saturday Night Live” has set Maya Rudolph to host its May 11 episode alongside Vampire Weekend as musical guest, while the May 18 episode will by headlined by Jake Gyllenhaal with performances from Sabrina Carpenter.

Rudolph was an “SNL” cast member for seven seasons from 2000 to 2007 and returns to the late-night sketch show for her third time as host, having previously hosted in 2012 with musical guest Sleigh Bells and in 2021 alongside Jack Harlow. Currently, she stars in and executive produces the Apple TV+ comedy “Loot.” Besides “SNL,” her prominent credits include “Bridesmaids,” “Sisters,” “The Good Place” and “Big Mouth.”

Vampire Weekend is currently touring its fifth album, “Only God Was Above Us.” May 11 will mark the band’s fourth appearance on “SNL.” Their previous albums include “Vampire Weekend,” “Contra,” “Modern Vampires of the City” and “Father of the Bride.”


Gyllenhaal will host the “SNL” Season 49 finale as a third-time host, having previously led episodes in 2007 with the Shins as musical guest and in 2022 opposite Camila Cabello. The actor recently starred in Amazon Prime Video’s remake of the 1989 Patrick Swayze film “Road House” and will next be seen in Apple TV+’s “Presumed Innocent,” which he also executive produces. He is best known for films such as “Donnie Darko,” “Brokeback Mountain,” “Zodiac” and “Nightcrawler.”

Carpenter will be making her “SNL” debut. She recently released the single “Espresso,” her first to hit the Billboard Top 10. She gained popularity as a musician after first building a career as an actress, most prominently on Disney Channel shows “Girl Meets World.” Her previous albums include “Emails I Can’t Send,” “Singular: Act I,” “Singular: Act II,” “Evolution” and “Eyes Wide Open.”
